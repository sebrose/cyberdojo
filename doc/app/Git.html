def commit(dir, args)
  c1 = run(dir, 'commit', args)
  c2 = run(dir, 'gc', '--aggressive --quiet')
  c1 + c2
end

def init(dir, args)
  if !File.directory?(dir)
    raise "Cannot `cd #{dir};git init` because #{dir} does not exist"
  end
  run(dir, 'init', args)
end
